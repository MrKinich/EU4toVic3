#should_be_naked_trigger
#should_be_naked_character_trigger

should_be_naked_trigger = {
	exists = scope:pop
	scope:pop = {
		trigger_if = {
			limit = {
				religion = { is_target_in_global_variable_list = { name = global_popular_nudist_religions target = this } }
			}
			NOT = { is_pop_type = academics }
		}
		trigger_else_if = {
			limit = { is_pop_type = clergymen }
			religion = { is_target_in_global_variable_list = { name = global_clerical_nudist_religions target = this } }
		}
		trigger_else_if = {
			limit = { is_pop_type = officers }
			religion = { is_target_in_global_variable_list = { name = global_clerical_nudist_religions target = this } }
			has_state_religion = yes
			owner = {
				ig:ig_armed_forces ?= {
					OR = {
						has_ideology = ideology:ideology_pious
						has_ideology = ideology:ideology_corporatist
					}
				}
			}
		}
		trigger_else = {
			OR = {
				is_pop_type = farmers
				is_pop_type = peasants
				is_pop_type = laborers
				is_pop_type = slaves
			}
			OR = {
				culture = cu:amazonian
				culture = cu:nuba
				culture = cu:dinka
			}
		}
	}
}

should_be_naked_character_trigger = {
	should_show_nudity = yes
	exists = scope:character
	scope:character = {
    	NOR = {
			has_ideology = ideology:ideology_nihilist
			has_ideology = ideology:ideology_communist
			has_ideology = ideology:ideology_positivist
		}
		trigger_if = {
			limit = {
				religion = { is_target_in_global_variable_list = { name = global_popular_nudist_religions target = this } }
			}
			NOT = { is_interest_group_type = ig_intelligentsia }
		}
		trigger_else_if = {
			limit = {
				religion = { is_target_in_global_variable_list = { name = global_clerical_nudist_religions target = this } }
			}
			OR = {
				has_ideology = ideology:ideology_theocrat
				is_interest_group_type = ig_devout
				trigger_if = {
					limit = {
						owner = { has_law_or_variant = law_type:law_theocracy }
					}
					is_ruler = yes
				}
				trigger_if = {
					limit = {
						owner = {
							religion = { is_target_in_global_variable_list = { name = global_clerical_nudist_religions target = this } }
							ig:ig_armed_forces ?= {
								OR = {
									has_ideology = ideology:ideology_pious
									has_ideology = ideology:ideology_corporatist
								}
							}
						}
					}
					OR = {
						is_interest_group_type = ig_armed_forces
						is_commander = yes
					}
				}
			}
		}
		trigger_else = { always = no }
	}
}
